const express = require('express'); const bodyParser = require('body-parser'); const { Configuration, OpenAIApi } = require('openai'); const app = express(); app.use(bodyParser.json()); const configuration = new Configuration({ apiKey: process.env.OPENAI_API_KEY, }); const openai = new OpenAIApi(configuration); app.post('/api/sentence-test', async (req, res) => { const { text } = req.body; if (!text) { return res.status(400).json({ error: 'Text is required' }); } try { const completion = await openai.createChatCompletion({ model: 'gpt-4', messages: [ { role: 'system', content: 'You are a helpful assistant that generates sentence test questions from a given passage.', }, { role: 'user', content: `Please generate sentence test questions from the following passage in JSON format: ${text} The JSON should be an array of items with fields: id, question, note (optional).` }, ], temperature: 0.7, }); const raw = completion.choices[0].message.content; // Remove any Markdown code fences (```â€¦```) const cleaned = raw.replace(/```[\s\S]*?```/g, '').trim(); let items; try { items = JSON.parse(cleaned); } catch (parseErr) { console.error('JSON parse error in sentence-test:', parseErr, 'Cleaned:', cleaned); return res.status(500).json({ error: 'Failed to parse JSON from AI response' }); } res.json({ items }); } catch (err) { console.error('Error in sentence-test:', err); res.status(500).json({ error: 'Internal server error' }); } }); const PORT = process.env.PORT || 3000; app.listen(PORT, () => { console.log(`Server listening on port ${PORT}`); });
